import os
import pandas as pd
import psycopg2
from dotenv import load_dotenv, find_dotenv
from sqlalchemy import create_engine
from sqlalchemy.engine import URL

# Locate and load the local .env file.
# This keeps sensitive PostgreSQL credentials out of the source code,
# GitHub repository, and rendered HTML report.
env_path = find_dotenv(usecwd=True)
load_dotenv(env_path)

# Read PostgreSQL connection settings from secured environment variables.
DB_USER = os.getenv("DB_USER")
DB_PASSWORD = os.getenv("DB_PASSWORD")
DB_HOST = os.getenv("DB_HOST")
DB_PORT = os.getenv("DB_PORT")
DB_NAME = os.getenv("DB_NAME")

# Validate that all required PostgreSQL environment variables are available.
required_variables = {
    "DB_USER": DB_USER,
    "DB_PASSWORD": DB_PASSWORD,
    "DB_HOST": DB_HOST,
    "DB_PORT": DB_PORT,
    "DB_NAME": DB_NAME
}

missing_variables = [
    variable_name
    for variable_name, variable_value in required_variables.items()
    if variable_value is None or variable_value == ""
]

if missing_variables:
    raise ValueError(f"Missing required environment variables: {missing_variables}")

# Build the PostgreSQL connection safely.
# URL.create prevents special characters in the password from breaking the connection string.
database_url = URL.create(
    drivername="postgresql+psycopg2",
    username=DB_USER,
    password=DB_PASSWORD,
    host=DB_HOST,
    port=int(DB_PORT),
    database=DB_NAME
)

# Create the SQLAlchemy engine used for pandas and PostgreSQL loading.
engine = create_engine(database_url)

# Establish a direct PostgreSQL connection test using psycopg2.
conn = psycopg2.connect(
    host=DB_HOST,
    port=DB_PORT,
    database=DB_NAME,
    user=DB_USER,
    password=DB_PASSWORD
)

print("PostgreSQL connection established successfully.")

Analyzing States with the Highest Number of Operational Schools

To begin the analytical portion of the project, I query PostgreSQL to identify which states have the highest total number of operational schools.

This query joins the state dimension table with the district school counts fact table and aggregates the results at the state level.

# SQL query to identify states with the highest operational school counts.
top_states_query = """
SELECT
    ds.state_name,
    SUM(f.operational_school_count) AS total_operational_schools
FROM public.fact_district_school_counts f
JOIN public.dim_states ds
    ON f.state_code = ds.state_code
GROUP BY ds.state_name
ORDER BY total_operational_schools DESC
LIMIT 10;
"""

Analyzing Average Operational Schools per District by State

To compare state-level district structures, I calculate the average number of operational schools per district for each state.

This helps show which states tend to have larger or smaller district-level school systems.

# SQL query to calculate the average number of operational schools per district by state.
avg_schools_per_district_query = """
SELECT
    ds.state_name,
    ROUND(AVG(f.operational_school_count)::numeric, 2) AS avg_operational_schools_per_district
FROM public.fact_district_school_counts f
JOIN public.dim_states ds
    ON f.state_code = ds.state_code
GROUP BY ds.state_name
ORDER BY avg_operational_schools_per_district DESC
LIMIT 10;
"""
